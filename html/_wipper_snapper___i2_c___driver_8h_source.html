67  int propertyIdx = 0;

69  switch (

71  case wippersnapper_i2c_v1_SensorType_SENSOR_TYPE_AMBIENT_TEMPERATURE:
72  enableSensorAmbientTemperature();
73  setSensorAmbientTemperaturePeriod(

75  break;

77  enableSensorRelativeHumidity();

80  break;
81  case wippersnapper_i2c_v1_SensorType_SENSOR_TYPE_PRESSURE:
82  enableSensorPressure();
83  setSensorPressurePeriod(

87  enableSensorCO2();

89  msgDeviceInitReq->i2c_device_properties[propertyIdx].sensor_period);
